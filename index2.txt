async function initDB() {
  return new Promise((resolve, reject) => {
    const req = indexedDB.open('LamsatSuppliersDB', 1);
    req.onerror = () => reject(req.error);
    req.onsuccess = () => { db = req.result; resolve(); };
    req.onupgradeneeded = (e) => {
      const d = e.target.result;
      if (!d.objectStoreNames.contains('state')) d.createObjectStore('state');
    };
  });
}

async function saveState() {
  if (!db) return;
  return new Promise((resolve) => {
    try {
      const tx = db.transaction(['state'], 'readwrite');
      tx.objectStore('state').put(state, 'app');
      tx.oncomplete = () => {
        document.getElementById('infoLastSave').textContent = new Date().toLocaleTimeString('ar-SA');
        resolve();
      };
      tx.onerror = () => resolve();
    } catch (e) { console.error(e); resolve(); }
  });
}

async function loadState() {
  if (!db) return;
  return new Promise((resolve) => {
    try {
      const tx = db.transaction(['state'], 'readonly');
      const req = tx.objectStore('state').get('app');
      req.onsuccess = () => {
        if (req.result) {
          state = req.result;
          if (!state.categories || state.categories.length === 0) state.categories = DEFAULT_CATEGORIES;
          if (!state.suppliers) state.suppliers = [];
          if (!state.orders) state.orders = [];
        } else {
          state.categories = DEFAULT_CATEGORIES;
          state.suppliers = [];
          state.orders = [];
        }
        resolve();
      };
      req.onerror = () => resolve();
    } catch (e) { resolve(); }
  });
}

function showToast(msg, type = 'info') {
  const c = document.getElementById('toastContainer');
  const t = document.createElement('div');
  t.className = 'toast';
  const colors = { success: 'var(--success)', error: 'var(--danger)', info: 'var(--primary)', warning: 'var(--warning)' };
  const icons = { success: 'check-circle', error: 'times-circle', info: 'info-circle', warning: 'exclamation-circle' };
  t.style.borderRightColor = colors[type];
  t.innerHTML = `<i class="fas fa-${icons[type]}" style="color:${colors[type]}; font-size:1.2rem;"></i><span>${msg}</span>`;
  c.appendChild(t);
  setTimeout(() => t.remove(), 3500);
}

function switchTab(tabId, btn) {
  document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));
  document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
  document.getElementById(tabId + 'Tab').classList.add('active');
  btn.classList.add('active');
  if (tabId === 'dashboard') renderDashboard();
  else if (tabId === 'suppliers') renderSuppliers();
  else if (tabId === 'orders') renderOrders();
  else if (tabId === 'tracking') renderTracking();
  else if (tabId === 'categories') renderCategories();
  else if (tabId === 'settings') updateSettingsInfo();
}

function renderSuppliers() {
  const search = (document.getElementById('supplierSearch')?.value || '').toLowerCase().trim();
  const statusFilter = document.getElementById('supplierStatusFilter')?.value || '';
  const catFilter = document.getElementById('supplierCategoryFilter')?.value || '';
  
  let filtered = state.suppliers.filter(s => {
    if (statusFilter && s.status !== statusFilter) return false;
    if (catFilter && !(s.assignedCategories || []).includes(catFilter)) return false;
    if (search) {
      const hay = (s.name + ' ' + s.contact + ' ' + s.email + ' ' + (s.address || '')).toLowerCase();
      const catNames = (s.assignedCategories || []).map(id => {
        const c = state.categories.find(x => x.id === id);
        return c ? c.name : '';
      }).join(' ').toLowerCase();
      if (!hay.includes(search) && !catNames.includes(search)) return false;
    }
    return true;
  });

  const grid = document.getElementById('suppliersGrid');
  if (filtered.length === 0) {
    grid.innerHTML = `<div class="empty-state" style="grid-column:1/-1;"><i class="fas fa-users-slash"></i><div class="empty-state-title">لا يوجد مورّدون</div><div>اضغط "مورد جديد" لإضافة أول مورد</div></div>`;
    return;
  }

  grid.innerHTML = filtered.map(s => {
    const ordersCount = state.orders.filter(o => o.supplierId === s.id).length;
    const activeOrders = state.orders.filter(o => o.supplierId === s.id && o.status !== 'delivered' && o.status !== 'cancelled').length;
    const assignedCats = (s.assignedCategories || []).map(id => state.categories.find(c => c.id === id)).filter(Boolean);
    const initial = (s.name || '?').charAt(0);
    
    return `
      <div class="supplier-card">
        <div class="supplier-header">
          <div class="supplier-avatar">${initial}</div>
          <div class="supplier-info">
            <div class="supplier-name">${s.name}</div>
            <div class="supplier-contact"><i class="fas fa-phone"></i> ${s.contact || 'غير محدد'}</div>
          </div>
          <span class="supplier-status ${s.status === 'active' ? 'status-active' : 'status-inactive'}">
            ${s.status === 'active' ? 'نشط' : 'غير نشط'}
          </span>
        </div>
        <div class="supplier-stats">
          <div class="supplier-stat">
            <div class="supplier-stat-label">المجموعات</div>
            <div class="supplier-stat-value">${assignedCats.length}</div>
          </div>
          <div class="supplier-stat">
            <div class="supplier-stat-label">الأوردرات</div>
            <div class="supplier-stat-value">${ordersCount}</div>
          </div>
        </div>
        <div class="supplier-categories">
          ${assignedCats.slice(0, 5).map(c => `<span class="category-chip">${c.name}</span>`).join('')}
          ${assignedCats.length > 5 ? `<span class="category-chip">+${assignedCats.length - 5}</span>` : ''}
        </div>
        <div class="supplier-actions">
          <button class="btn btn-sm btn-primary" onclick="openOrderWizard('${s.id}')"><i class="fas fa-plus"></i> أمر جديد</button>
          <button class="btn btn-sm btn-secondary" onclick="openCategoriesModal('${s.id}')"><i class="fas fa-layer-group"></i> المجموعات</button>
          <button class="btn btn-sm btn-secondary" onclick="editSupplier('${s.id}')"><i class="fas fa-edit"></i></button>
          <button class="btn btn-sm btn-danger" onclick="deleteSupplier('${s.id}')"><i class="fas fa-trash"></i></button>
        </div>
      </div>
    `;
  }).join('');
}

function openSupplierModal(id = null) {
  document.getElementById('supplierId').value = id || '';
  document.getElementById('supplierModalTitle').textContent = id ? 'تعديل مورد' : 'إضافة مورد';
  if (id) {
    const s = state.suppliers.find(x => x.id === id);
    if (s) {
      document.getElementById('supplierName').value = s.name || '';
      document.getElementById('supplierContact').value = s.contact || '';
      document.getElementById('supplierEmail').value = s.email || '';
      document.getElementById('supplierAddress').value = s.address || '';
      document.getElementById('supplierStatus').value = s.status || 'active';
      document.getElementById('supplierNotes').value = s.notes || '';
    }
  } else {
    ['supplierName','supplierContact','supplierEmail','supplierAddress','supplierNotes'].forEach(i => document.getElementById(i).value = '');
    document.getElementById('supplierStatus').value = 'active';
  }

}

function editSupplier(id) { openSupplierModal(id); }

function saveSupplier() {
  const id = document.getElementById('supplierId').value;
  const name = document.getElementById('supplierName').value.trim();
  if (!name) { showToast('اسم المورد مطلوب', 'error'); return; }
  
  const data = {
    name,
    contact: document.getElementById('supplierContact').value.trim(),
    email: document.getElementById('supplierEmail').value.trim(),
    address: document.getElementById('supplierAddress').value.trim(),
    status: document.getElementById('supplierStatus').value,
    notes: document.getElementById('supplierNotes').value.trim()
  };
  
  if (id) {
    const idx = state.suppliers.findIndex(s => s.id === id);
    if (idx >= 0) {
      state.suppliers[idx] = { ...state.suppliers[idx], ...data, updatedAt: new Date().toISOString() };
      showToast('تم تحديث المورد', 'success');
    }
  } else {
    state.suppliers.push({
      id: genId('SUP'),
      ...data,
      assignedCategories: [],
      createdAt: new Date().toISOString()
    });
    showToast('تم إضافة المورد', 'success');
  }
  
  saveState();
  closeModal('supplierModal');
  renderSuppliers();
  renderStats();
  populateSupplierCategoryFilter();
}

function deleteSupplier(id) {
  if (!confirm('هل أنت متأكد من حذف هذا المورد؟ سيتم حذف كل أوردراته أيضاً.')) return;
  state.suppliers = state.suppliers.filter(s => s.id !== id);
  state.orders = state.orders.filter(o => o.supplierId !== id);
  saveState();
  renderSuppliers();
  renderStats();
  populateSupplierCategoryFilter();
  showToast('تم حذف المورد', 'success');
}

  document.getElementById('supplierStatusFilter').value = '';
  document.getElementById('supplierCategoryFilter').value = '';
  renderSuppliers();
}

function updateWizardUI() {
  const step = state.currentWizard.step;
  for (let i = 1; i <= 4; i++) {
    const el = document.getElementById('wizardStep' + i);
    const stepEl = document.getElementById('step' + i);
    el.style.display = i === step ? 'block' : 'none';
    stepEl.classList.remove('active', 'completed');
    if (i < step) stepEl.classList.add('completed');
    else if (i === step) stepEl.classList.add('active');
  }
  document.getElementById('wizardPrevBtn').style.display = step > 1 ? 'inline-flex' : 'none';
  document.getElementById('wizardNextBtn').style.display = step < 4 ? 'inline-flex' : 'none';
  document.getElementById('wizardSaveBtn').style.display = step === 4 ? 'inline-flex' : 'none';
  
  if (step === 2) renderOrderCategories();
  if (step === 3) renderAllocationGroups();
  if (step === 4) renderOrderReview();
}

function wizardNext() {
  const step = state.currentWizard.step;
  if (step === 1) {
    if (!state.currentWizard.supplierId) { showToast('اختر المورد أولاً', 'error'); return; }
    const s = state.suppliers.find(x => x.id === state.currentWizard.supplierId);
    if (!s.assignedCategories || s.assignedCategories.length === 0) {
      showToast('هذا المورد ليس لديه مجموعات مخصصة. قم بتعيين مجموعات أولاً.', 'error');
      return;
    }
  } else if (step === 2) {
    if (state.currentWizard.selectedCategories.length === 0) {
      showToast('اختر مجموعة واحدة على الأقل', 'error'); return;
    }
  } else if (step === 3) {
    collectAllocations();
  }
  state.currentWizard.step = Math.min(4, step + 1);
  updateWizardUI();
}

function wizardPrev() {
  if (state.currentWizard.step === 3) collectAllocations();
  state.currentWizard.step = Math.max(1, state.currentWizard.step - 1);
  updateWizardUI();
}

function renderOrderCategories() {
  const s = state.suppliers.find(x => x.id === state.currentWizard.supplierId);
  const assigned = s?.assignedCategories || [];
  const selected = state.currentWizard.selectedCategories;
  
  const cats = state.categories.filter(c => assigned.includes(c.id));
  document.getElementById('orderCategoriesList').innerHTML = cats.map(c => `
    <label class="category-checkbox ${selected.includes(c.id) ? 'checked' : ''}">
      <input type="checkbox" ${selected.includes(c.id) ? 'checked' : ''} onchange="toggleOrderCategory('${c.id}', this)">
      <span class="category-checkbox-label">${c.name}</span>
      <span style="margin-right:auto; font-size:0.75rem; color:var(--text-secondary);">(${c.subs.length})</span>
    </label>
  `).join('');
}

function toggleOrderCategory(id, cb) {
  const label = cb.closest('.category-checkbox');
  label.classList.toggle('checked', cb.checked);
  if (cb.checked) {
    if (!state.currentWizard.selectedCategories.includes(id)) state.currentWizard.selectedCategories.push(id);
  } else {
    state.currentWizard.selectedCategories = state.currentWizard.selectedCategories.filter(x => x !== id);
    delete state.currentWizard.allocations[id];
  }
}

        </div>

          <table class="data-table">

function updateAllocationTarget(catId, val) {
  if (!state.currentWizard.allocations[catId]) state.currentWizard.allocations[catId] = { target: 0, items: {} };
  state.currentWizard.allocations[catId].target = parseInt(val) || 0;
  updateAllocationSummary(catId);
}

function updateSubQty(catId, subId, val) {
  if (!state.currentWizard.allocations[catId]) state.currentWizard.allocations[catId] = { target: 0, items: {} };
  if (!state.currentWizard.allocations[catId].items[subId]) state.currentWizard.allocations[catId].items[subId] = { qty: 0, note: '' };
  state.currentWizard.allocations[catId].items[subId].qty = parseInt(val) || 0;
  updateAllocationSummary(catId);
}

function updateSubNote(catId, subId, val) {
  if (!state.currentWizard.allocations[catId]) state.currentWizard.allocations[catId] = { target: 0, items: {} };
  if (!state.currentWizard.allocations[catId].items[subId]) state.currentWizard.allocations[catId].items[subId] = { qty: 0, note: '' };
  state.currentWizard.allocations[catId].items[subId].note = val;
}

function updateAllocationSummary(catId) {
  const a = state.currentWizard.allocations[catId] || { target: 0, items: {} };
  const allocated = Object.values(a.items || {}).reduce((s, i) => s + (parseInt(i.qty) || 0), 0);
  const target = parseInt(a.target) || 0;
  const shortage = target - allocated;
  const progress = target > 0 ? Math.round((allocated / target) * 100) : 0;
  const progressClass = progress >= 100 ? 'success' : progress >= 50 ? 'warning' : 'danger';
  
  const el = document.getElementById('summary-' + catId);
  if (el) {
    el.innerHTML = `
      <div class="allocation-summary-item">
        <div class="allocation-summary-label">المستهدف</div>
        <div class="allocation-summary-value" style="color:var(--info);">${target.toLocaleString()}</div>
      </div>
      <div class="allocation-summary-item">
        <div class="allocation-summary-label">المسجل</div>
        <div class="allocation-summary-value" style="color:var(--success);">${allocated.toLocaleString()}</div>
      </div>
      <div class="allocation-summary-item">
        <div class="allocation-summary-label">العجز</div>
        <div class="allocation-summary-value" style="color:${shortage > 0 ? 'var(--danger)' : 'var(--success)'};">${shortage.toLocaleString()}</div>
      </div>
      <div class="allocation-summary-item">
        <div class="allocation-summary-label">الإنجاز</div>
        <div class="allocation-summary-value" style="color:var(--primary);">${progress}%</div>
        <div class="progress-bar" style="margin-top:0.3rem;"><div class="progress-fill ${progressClass}" style="width:${Math.min(100, progress)}%;"></div></div>
      </div>
    `;
  }
}

            <td style="color:${(totalTarget-totalAllocated) > 0 ? 'var(--danger)' : 'var(--success)'};">${(totalTarget-totalAllocated).toLocaleString()}</td>
            <td><strong style="color:var(--primary); font-size:1.1rem;">${overallProgress}%</strong></td>
          </tr>
        </tfoot>

  `;

function saveOrder() {
  const s = state.suppliers.find(x => x.id === state.currentWizard.supplierId);
  if (!s) return;
  
  const items = state.currentWizard.selectedCategories.map(catId => {
    const cat = state.categories.find(c => c.id === catId);
    const a = state.currentWizard.allocations[catId] || { target: 0, items: {} };
    const allocated = Object.values(a.items || {}).reduce((sum, i) => sum + (parseInt(i.qty) || 0), 0);
    const target = parseInt(a.target) || 0;
    return {
      mainCategoryId: catId,
      mainCategoryName: cat.name,
      target,
      subAllocations: a.items || {},
      totalAllocated: allocated,
      shortfall: target - allocated,
      progress: target > 0 ? Math.round((allocated / target) * 100) : 0
    };
  });
  
  const totals = {
    targetSum: items.reduce((s, i) => s + i.target, 0),
    allocatedSum: items.reduce((s, i) => s + i.totalAllocated, 0),
    shortfallSum: items.reduce((s, i) => s + i.shortfall, 0),
    overallProgress: 0
  };
  totals.overallProgress = totals.targetSum > 0 ? Math.round((totals.allocatedSum / totals.targetSum) * 100) : 0;
  
  const order = {
    id: genId('ORD'),
    supplierId: s.id,
    supplierName: s.name,
    orderDate: new Date().toISOString(),
    status: 'draft',
    notes: '',
    items,
    totals
  };
  
  state.orders.push(order);
  saveState();
  closeModal('orderModal');
  renderOrders();
  renderStats();
  showToast('تم حفظ الأمر بنجاح', 'success');
}

function renderTracking() {
  const search = (document.getElementById('trackingSearch')?.value || '').toLowerCase().trim();
  const suppliers = state.suppliers.filter(s => !search || s.name.toLowerCase().includes(search));
  
  const content = document.getElementById('trackingContent');
  if (suppliers.length === 0) {
    content.innerHTML = `<div class="empty-state"><i class="fas fa-chart-line"></i><div class="empty-state-title">لا توجد بيانات</div></div>`;
    return;
  }
  
  content.innerHTML = suppliers.map(s => {
    const orders = state.orders.filter(o => o.supplierId === s.id);
    if (orders.length === 0) return '';
    
    const totalTarget = orders.reduce((sum, o) => sum + (o.totals?.targetSum || 0), 0);
    const totalAllocated = orders.reduce((sum, o) => sum + (o.totals?.allocatedSum || 0), 0);
    const totalShortage = totalTarget - totalAllocated;
    const progress = totalTarget > 0 ? Math.round((totalAllocated / totalTarget) * 100) : 0;
    const progressClass = progress >= 100 ? 'success' : progress >= 50 ? 'warning' : 'danger';
    const status = progress >= 100 ? 'مكتمل' : progress > 0 ? 'تحت التنفيذ' : 'لم يبدأ';
    const statusColor = progress >= 100 ? 'badge-success' : progress > 0 ? 'badge-warning' : 'badge-info';
    
    // Aggregate by category
    const catMap = {};
    orders.forEach(o => {
      o.items.forEach(item => {
        if (!catMap[item.mainCategoryId]) {
          catMap[item.mainCategoryId] = { name: item.mainCategoryName, target: 0, allocated: 0 };
        }
        catMap[item.mainCategoryId].target += item.target;
        catMap[item.mainCategoryId].allocated += item.totalAllocated;
      });
    });
    
    return `
      <div class="card" style="margin-bottom:1rem;">
        <div class="card-header">
          <h3 class="card-title"><i class="fas fa-user-tie"></i> ${s.name}</h3>
          <span class="badge ${statusColor}">${status}</span>
        </div>
        <div class="allocation-summary" style="margin-bottom:1rem;">
          <div class="allocation-summary-item">
            <div class="allocation-summary-label">المستهدف</div>
            <div class="allocation-summary-value" style="color:var(--info);">${totalTarget.toLocaleString()}</div>
          </div>
          <div class="allocation-summary-item">
            <div class="allocation-summary-label">المسجل</div>
            <div class="allocation-summary-value" style="color:var(--success);">${totalAllocated.toLocaleString()}</div>
          </div>
          <div class="allocation-summary-item">
            <div class="allocation-summary-label">العجز</div>
            <div class="allocation-summary-value" style="color:${totalShortage > 0 ? 'var(--danger)' : 'var(--success)'};">${totalShortage.toLocaleString()}</div>
          </div>
          <div class="allocation-summary-item">
            <div class="allocation-summary-label">الإنجاز</div>
            <div class="allocation-summary-value" style="color:var(--primary);">${progress}%</div>
            <div class="progress-bar" style="margin-top:0.3rem;"><div class="progress-fill ${progressClass}" style="width:${Math.min(100, progress)}%;"></div></div>
          </div>
        </div>
        <div class="table-wrapper">
          <table class="data-table">
            <thead>
              <tr><th>المجموعة</th><th>المستهدف</th><th>المسجل</th><th>العجز</th><th>الإنجاز</th></tr>

    `;
  }).join('');
}

      <h3 style="color:var(--primary); margin-bottom:1rem;"><i class="fas fa-layer-group"></i> تقرير المجموعات</h3>
      <div class="table-wrapper"><table class="data-table">
        <thead><tr><th>المجموعة</th><th>عدد الأوردرات</th><th>المستهدف</th><th>المسجل</th><th>العجز</th><th>الإنجاز</th></tr></thead>
        <tbody>
          ${Object.values(catMap).sort((a, b) => b.target - a.target).map(c => {
            const p = c.target > 0 ? Math.round((c.allocated / c.target) * 100) : 0;
            return `
              <tr>
                <td><strong>${c.name}</strong></td>
                <td>${c.ordersCount}</td>
                <td>${c.target.toLocaleString()}</td>
                <td>${c.allocated.toLocaleString()}</td>
                <td style="color:${(c.target-c.allocated) > 0 ? 'var(--danger)' : 'var(--success)'};">${(c.target-c.allocated).toLocaleString()}</td>
                <td><strong style="color:var(--primary);">${p}%</strong></td>
              </tr>
            `;
          }).join('') || '<tr><td colspan="6" style="text-align:center; padding:2rem;">لا بيانات</td></tr>'}
        </tbody>
      </table></div>
    `;
  } else if (type === 'shortage') {
    const shortages = [];
    state.orders.forEach(o => {
      o.items.forEach(item => {
        if (item.shortfall > 0) {
          shortages.push({
            order: o.id,
            supplier: o.supplierName,
            category: item.mainCategoryName,
            target: item.target,
            allocated: item.totalAllocated,
            shortfall: item.shortfall
          });
        }
      });
    });
    shortages.sort((a, b) => b.shortfall - a.shortfall);
    html = `
      <h3 style="color:var(--primary); margin-bottom:1rem;"><i class="fas fa-exclamation-triangle"></i> تقرير العجز</h3>
      <div class="table-wrapper"><table class="data-table">
        <thead><tr><th>المورد</th><th>المجموعة</th><th>المستهدف</th><th>المسجل</th><th>العجز</th></tr></thead>
        <tbody>
          ${shortages.map(s => `
            <tr>
              <td>${s.supplier}</td>
              <td><strong>${s.category}</strong></td>
              <td>${s.target.toLocaleString()}</td>
              <td>${s.allocated.toLocaleString()}</td>
              <td style="color:var(--danger); font-weight:700;">${s.shortfall.toLocaleString()}</td>
            </tr>
          `).join('') || '<tr><td colspan="5" style="text-align:center; padding:2rem; color:var(--success);">✓ لا يوجد عجز - كل شيء مكتمل</td></tr>'}
        </tbody>
      </table></div>
    `;
  }
  
  content.innerHTML = html;
}

function handleImport(event) {
  const file = event.target.files[0];
  if (!file) return;
  showLoading(true, 'جاري استيراد المجموعات...');
  
  const reader = new FileReader();
  reader.onload = (e) => {
    try {
      const data = new Uint8Array(e.target.result);
      const wb = XLSX.read(data, { type: 'array' });
      const ws = wb.Sheets[wb.SheetNames[0]];
      const rows = XLSX.utils.sheet_to_json(ws, { header: 1, defval: '' });
      
      if (rows.length < 2) { showToast('الملف فارغ', 'error'); showLoading(false); return; }
      
      // Detect columns
      const headers = rows[0].map(h => String(h).toLowerCase().trim());
      const idIdx = headers.findIndex(h => h.includes('المجموعة') && !h.includes('فرعية') && !h.includes('اسم'));
      const nameIdx = headers.findIndex(h => h.includes('اسم') && h.includes('رئيسية'));
      const subIdx = headers.findIndex(h => h.includes('فرعية'));
      
      if (nameIdx < 0 || subIdx < 0) {
        showToast('تعذر التعرف على الأعمدة. تأكد من وجود: اسم المجموعة الرئيسية، اسم المجموعة الفرعية', 'error');
        showLoading(false);
        return;
      }
      
      const catMap = {};
      for (let i = 1; i < rows.length; i++) {
        const row = rows[i];
        const id = String(row[idIdx >= 0 ? idIdx : 0] || '').trim();
        const name = String(row[nameIdx] || '').trim();
        const sub = String(row[subIdx] || '').trim();
        if (!name || !sub) continue;
        const key = id || name;
        if (!catMap[key]) catMap[key] = { id: id || key, name, subs: [] };
        if (!catMap[key].subs.includes(sub)) catMap[key].subs.push(sub);
      }
      
      const imported = Object.values(catMap);
      if (imported.length === 0) {
        showToast('لم يتم العثور على بيانات', 'error');
      } else {
        state.categories = imported;
        saveState();
        renderStats();
        populateSupplierCategoryFilter();
        showToast(`تم استيراد ${imported.length} مجموعة رئيسية`, 'success');
      }
    } catch (err) {
      console.error(err);
      showToast('خطأ في الاستيراد: ' + err.message, 'error');
    }
    showLoading(false);
  };
  reader.readAsArrayBuffer(file);
  event.target.value = '';
}

function exportToExcel() {
  try {
    const wb = XLSX.utils.book_new();
    
    // Suppliers sheet
    const supData = state.suppliers.map(s => ({
      'الرقم': s.id,
      'الاسم': s.name,
      'الجوال': s.contact,
      'البريد': s.email,
      'العنوان': s.address,
      'الحالة': s.status,
      'المجموعات': (s.assignedCategories || []).join(' | ')
    }));
    if (supData.length > 0) {
      const ws1 = XLSX.utils.json_to_sheet(supData);
      XLSX.utils.book_append_sheet(wb, ws1, 'الموردون');
    }
    
    // Orders sheet
    const ordData = state.orders.map(o => ({
      'رقم الأمر': o.id,
      'المورد': o.supplierName,
      'التاريخ': new Date(o.orderDate).toLocaleDateString('ar-SA'),
      'الحالة': o.status,
      'المستهدف': o.totals.targetSum,
      'المسجل': o.totals.allocatedSum,
      'العجز': o.totals.shortfallSum,
      'الإنجاز %': o.totals.overallProgress
    }));
    if (ordData.length > 0) {
      const ws2 = XLSX.utils.json_to_sheet(ordData);
      XLSX.utils.book_append_sheet(wb, ws2, 'الأوردرات');
    }
    
    // Categories sheet
    const catData = [];
    state.categories.forEach(c => {
      c.subs.forEach(s => catData.push({ 'الرقم': c.id, 'المجموعة الرئيسية': c.name, 'المجموعة الفرعية': s }));
    });
    if (catData.length > 0) {
      const ws3 = XLSX.utils.json_to_sheet(catData);
      XLSX.utils.book_append_sheet(wb, ws3, 'المجموعات');
    }
    
    XLSX.writeFile(wb, `Suppliers_Report_${new Date().toLocaleDateString('ar-SA').replace(/\//g, '-')}.xlsx`);
    showToast('تم التصدير بنجاح', 'success');
  } catch (err) {
    showToast('خطأ في التصدير', 'error');
  }
}

function backupData() {
  const data = JSON.stringify(state, null, 2);
  const blob = new Blob([data], { type: 'application/json' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `backup_${new Date().toLocaleDateString('ar-SA').replace(/\//g, '-')}.json`;
  a.click();
  URL.revokeObjectURL(url);
  showToast('تم تحميل النسخة الاحتياطية', 'success');
}

function restoreData(event) {
  const file = event.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = (e) => {
    try {
      const data = JSON.parse(e.target.result);
      if (data.categories && data.suppliers && data.orders) {
        state = data;
        saveState();
        renderAll();
        showToast('تم استعادة البيانات بنجاح', 'success');
      } else {
        showToast('ملف غير صالح', 'error');
      }
    } catch (err) {
      showToast('خطأ في قراءة الملف', 'error');
    }
  };
  reader.readAsText(file);
  event.target.value = '';
}
